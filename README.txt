git init                  -Iniciar git
git clone URL             -Copiar repositorio
git remote add origin URL -Permitir actualizar repositorio

git status                -Para ver cambios en la carpeta local
git add -A                -Para añadir archivos y luego subirlos
git commit -m MENSAJE     -Para confirmar archivos añadidos
git push origin master    -Para subir archivos confirmados
git pull		  -Para descargar cambios
